import useBaseUrl from "@docusaurus/core/lib/client/exports/useBaseUrl";

# \<SVG />

## Assignable props
<AssignableProps
  data={[
    {name: 'src', value: 'string', required: true, definition: 'Source path'},
    {name: 'color', value: 'string', required: false, definition: 'Override svg color'},
    {name: 'height', value: 'number', required: false, definition: 'Height of the svg', default: '32px'},
    {name: 'width', value: 'number', required: false, definition: 'Width of the svg', default: '32px'},
  ]}
/>

## Example usage

### Default svg
<SVG src={useBaseUrl('/img/example.svg#question')}/>

&NewLine;

```tsx
<SVG src={'path/to/file.svg'}/>
```

### Large svg
<SVG src={useBaseUrl('/img/example.svg#question')} height={64} width={64}/>

&NewLine;

```tsx
<SVG src={'path/to/file.svg'} height={64} width={64}/>
```

### Custom color
<SVG src={useBaseUrl('/img/example.svg#question')} color={'#0047CC'}/>

&NewLine;

```tsx
<SVG src={'path/to/file.svg'} color={'#0047CC'}/>
```
