# \<Radio />

## Assignable props
<AssignableProps
  data={[
    {name: 'options', value: 'RadioOption[]', required: true, definition: 'Radio options'},
    {name: 'value', value: 'string', required: true, definition: 'Current value'},
    {name: 'valueChanged', value: '(value: string) => void', required: true, definition: 'Update the current value'},
    {name: 'color', value: 'string', required: false, definition: 'Set the check color'},
    {name: 'disabled', value: 'boolean', required: false, definition: 'disable all options'},
  ]}
/>

## RadioOption
<AssignableProps
  data={[
    {name: 'name', value: 'string', required: true, definition: 'name of the option'},
    {name: 'value', value: 'string', required: true, definition: 'value of the option'},
    {name: 'disabled', value: 'boolean', required: false, definition: 'Disable the option'},
  ]}
/>
